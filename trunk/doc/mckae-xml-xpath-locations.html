type Node_Test_Specification
  (Node_Test : Node_Tests := No_Node_Test) is
   record
      Name : Unbounded_String;
      case Node_Test is
         when QName_Node_Test =>
            Prefix : Unbounded_String;
         when No_Node_Test   |
           NCName_Node_Test   |
           Text_Node_Test    |
           Comment_Node_Test |
           Processing_Instruction_Node_Test |
           Node_Node_Test =>
            null;
      end case;
   end record;
